import time
import pyautogui
 
 
print('Press CTRL-C to End')
 
try:
     while True:
         pyautogui.moveRel(-100, 0)
         pyautogui.moveRel(100, 0)
         time.sleep(15)
except KeyboardInterrupt:
     print('\nEnded')

 #O programa meche o mouse automaticamente a cada 15s no caso mas isso pode ser alterado mudando o valor dentro da função time.sleep.
#Isso evita que o computador bloqueie sem voce desejar caso queira deixar ele aberto enquanto nao está perto dele  ou manter a tela acesa sem precisar ficar mechendo o mouse.
